cmake_minimum_required(VERSION 3.16)
project(EnhancedTerminalDemo VERSION 1.0.0 LANGUAGES CXX)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add compiler flags for better warnings and debugging
if(MSVC)
    add_compile_options(/W4)
else()
    add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# Create the executable
add_executable(enhanced_terminal_demo enhanced_terminal_demo.cpp)

# Platform-specific libraries
if(WIN32)
    target_link_libraries(enhanced_terminal_demo PRIVATE)
else()
    target_link_libraries(enhanced_terminal_demo PRIVATE)
endif()

# Set output directory
set_target_properties(enhanced_terminal_demo PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)

# Installation
install(TARGETS enhanced_terminal_demo
    RUNTIME DESTINATION bin
)
